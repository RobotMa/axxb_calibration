  649   //                double *b
  650   //                double *c
  651   //                double *s
  652   // Return Type  : void
  653   //
  654   static void eml_xrotg(double *a, double *b, double *c, double *s)
  655   {
  656     double roe;
  657     double absa;
  658     double absb;
  659     double scale;
  660     double ads;
  661     double bds;
  662     roe = *b;
  663     absa = fabs(*a);
  664     absb = fabs(*b);
  665     if (absa > absb) {
  666       roe = *a;
  667     }
  668   
  669     scale = absa + absb;
  670     if (scale == 0.0) {
  671       *s = 0.0;
  672       *c = 1.0;
  673       scale = 0.0;
  674       *b = 0.0;
  675     } else {
  676       ads = absa / scale;
  677       bds = absb / scale;
  678       scale *= sqrt(ads * ads + bds * bds);
  679       if (roe < 0.0) {
  680         scale = -scale;
  681       }
  682   
  683       *c = *a / scale;
  684       *s = *b / scale;
  685       if (absa > absb) {
  686         *b = *s;
  687       } else if (*c != 0.0) {
  688         *b = 1.0 / *c;
  689       } else {
  690         *b = 1.0;
  691       }
  692     }
  693   
  694     *a = scale;
  695   }

  703   static void eml_xscal(double a, double x[9], int ix0)

  705     int k;
  706     for (k = ix0; k <= ix0 + 2; k++) {
  707       x[k - 1] *= a;
  708     }

  712   // Arguments    : double x[9]
  713   //                int ix0
  714   //                int iy0
  715   // Return Type  : void
  716   //
  717   static void eml_xswap(double x[9], int ix0, int iy0)
  718   {
  719     int ix;
  720     int iy;
  721     int k;
  722     double temp;
  723     ix = ix0 - 1;
  724     iy = iy0 - 1;
  725     for (k = 0; k < 3; k++) {
  726       temp = x[ix];
  727       x[ix] = x[iy];
  728       x[iy] = temp;
  729       ix++;
  730       iy++;
  731     }
  732   }

  734   //
  735   // Arguments    : const emxArray_real_T *X
  736   //                double M_1[16]
  737   //                double M_hat[16]
  738   // Return Type  : void
  739   //
  740   void mean_Taylor_1st(const emxArray_real_T *X, double M_1[16], double M_hat[16])
  741   {
  742     double n;
  743     int i;
  744     emxArray_real_T *b_X;
  745     int i0;
  746     unsigned int u0;
  747     int i1;
  748     int i2;
  749     int loop_ub;
  750     double y;
  751     double b_M_hat[9];
  752     double V[9];
  753     double s[3];
  754     double U[9];
  755     double Rx[9];
  756     static const signed char iv0[4] = { 0, 0, 0, 1 };
  757   
  758     //  This function calculates the 1st order approximation of the mean of a
  759     //  bunch of matrices based on the Taylor expansion of the matrix logarithm
  760     //  and the definition of mean of a probability density function.
  761     //  Input: X is a 4 by 4*n rigid transformation matrices
  762     //  Output: M_T1 is the mean of the 1st order approximation of Taylor
  763     //  expansion
  764     //  Change of this m file doesn't automatically change the executable generated by  
  765     //  mean_Taylor_2nd.m
  766     n = (double)X->size[1] / 4.0;
  767     memset(&M_hat[0], 0, sizeof(double) << 4);
  768     i = 0;
  769     emxInit_real_T(&b_X, 2);
  770     while (i <= (int)n - 1) {
  771       i0 = (i << 2) + 1;
  772       u0 = (unsigned int)(1 + i) << 2;
  773       if ((unsigned int)i0 > u0) {
  774         i0 = 0;
  775         i1 = 0;
  776       } else {
  777         i0--;
  778         i1 = (int)u0;
  779       }
  780   
  781       i2 = b_X->size[0] * b_X->size[1];
  782       b_X->size[0] = 4;
  783       b_X->size[1] = i1 - i0;
  784       emxEnsureCapacity((emxArray__common *)b_X, i2, (int)sizeof(double));
  785       loop_ub = i1 - i0;
  786       for (i1 = 0; i1 < loop_ub; i1++) {
  787         for (i2 = 0; i2 < 4; i2++) {
  788           b_X->data[i2 + b_X->size[0] * i1] = X->data[i2 + X->size[0] * (i0 + i1)];
  789         }
  790       }
  791   
  792       for (i0 = 0; i0 < 4; i0++) {
  793         for (i1 = 0; i1 < 4; i1++) {
  794           M_hat[i1 + (i0 << 2)] += b_X->data[i1 + (i0 << 2)];
  795         }
  796       }
  797   
  798       i++;
  799     }
  800   
  801     emxFree_real_T(&b_X);
  802     y = 1.0 / n;
  803     for (i0 = 0; i0 < 16; i0++) {
  804       M_hat[i0] *= y;
  805     }
  806   
  807     //  Note that M_hat doesn't belong to SE(3)
  808     //  This function takes in a 4 by 4 matrix which has the form of
  809     //  [3 by 4; 0 0 0 n] and return a rigig transformation matrix by
  810     //  orthogonalizing the upper left 3 by 3 matrix into a rotation matrix
  811     //  opt = 1: Orthogonalize only the rotation part of M_hat
  812     //  opt = 2: SVD the whole M_hat to obtain a transformation matrix
  813     for (i0 = 0; i0 < 3; i0++) {
  814       for (i1 = 0; i1 < 3; i1++) {
  815         b_M_hat[i1 + 3 * i0] = M_hat[i1 + (i0 << 2)];
  816       }
  817     }
  818   
  819     eml_xgesvd(b_M_hat, U, s, V);
  820   
  821     //  The order of V and U might be flipped
  822     for (i0 = 0; i0 < 3; i0++) {
  823       for (i1 = 0; i1 < 3; i1++) {
  824         Rx[i0 + 3 * i1] = 0.0;
  825         for (i2 = 0; i2 < 3; i2++) {
  826           Rx[i0 + 3 * i1] += U[i0 + 3 * i2] * V[i1 + 3 * i2];
  827         }
  828       }
  829     }
  830   
  831     //      Rx = V*U';
  832     if (det(Rx) < 0.0) {
  833       for (i0 = 0; i0 < 3; i0++) {
  834         Rx[6 + i0] = -Rx[6 + i0];
  835       }
  836     }
  837   
  838     if (fabs(det(Rx) - 1.0) > 1.0E-8) {
  839       b_fprintf();
  840     }
  841   
  842     for (i0 = 0; i0 < 3; i0++) {
  843       for (i1 = 0; i1 < 3; i1++) {
  844         M_1[i1 + (i0 << 2)] = Rx[i1 + 3 * i0];
  845       }
  846     }
  847   
  848     for (i0 = 0; i0 < 3; i0++) {
  849       M_1[12 + i0] = M_hat[12 + i0];
  850     }
  851   
  852     for (i0 = 0; i0 < 4; i0++) {
  853       M_1[3 + (i0 << 2)] = iv0[i0];
  854     }
  855   }
